    <script>
        // --- THREE.JS BACKGROUND (Layer 2) ---
        let scene, camera, renderer, hearts = [];
        const canvas = document.querySelector('#three-canvas');

        function initThree() {
            scene = new THREE.Scene();
            camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
            renderer = new THREE.WebGLRenderer({ canvas, alpha: true, antialias: true });
            renderer.setSize(window.innerWidth, window.innerHeight);
            renderer.setPixelRatio(window.devicePixelRatio);

            const heartShape = new THREE.Shape();
            heartShape.moveTo( 25, 25 );
            heartShape.bezierCurveTo( 25, 25, 20, 0, 0, 0 );
            heartShape.bezierCurveTo( -30, 0, -30, 35, -30, 35 );
            heartShape.bezierCurveTo( -30, 55, -10, 77, 25, 95 );
            heartShape.bezierCurveTo( 60, 77, 80, 55, 80, 35 );
            heartShape.bezierCurveTo( 80, 35, 80, 0, 50, 0 );
            heartShape.bezierCurveTo( 35, 0, 25, 25, 25, 25 );

            const extrudeSettings = { depth: 8, bevelEnabled: true, bevelSegments: 2, steps: 2, bevelSize: 1, bevelThickness: 1 };
            const geometry = new THREE.ExtrudeGeometry(heartShape, extrudeSettings);
            
            for (let i = 0; i < 22; i++) {
                const material = new THREE.MeshPhongMaterial({ 
                    color: i % 2 === 0 ? 0xec4899 : 0xef4444,
                    shininess: 100 
                });
                const mesh = new THREE.Mesh(geometry, material);
                
                const scale = Math.random() * 0.015 + 0.005;
                mesh.scale.set(scale, scale, scale);
                mesh.rotation.z = Math.PI;
                mesh.rotation.x = Math.PI;
                
                mesh.position.set(
                    (Math.random() - 0.5) * 15,
                    (Math.random() - 0.5) * 15,
                    (Math.random() - 0.5) * 10
                );
                
                mesh.userData = {
                    speed: Math.random() * 0.01 + 0.005,
                    offset: Math.random() * Math.PI * 2
                };
                
                scene.add(mesh);
                hearts.push(mesh);
            }

            const light = new THREE.DirectionalLight(0xffffff, 1.5);
            light.position.set(1, 1, 5);
            scene.add(light);
            scene.add(new THREE.AmbientLight(0x404040, 2));

            camera.position.z = 8;
        }

        function animate() {
            requestAnimationFrame(animate);
            const time = Date.now() * 0.001;
            hearts.forEach(h => {
                h.position.y += Math.sin(time + h.userData.offset) * 0.005;
                h.rotation.y += h.userData.speed;
            });
            renderer.render(scene, camera);
        }

        // --- STEP NAVIGATION ---
        let currentStep = 1;
        function nav(step) {
            const outEl = document.getElementById(`step-${currentStep}`);
            const inEl = document.getElementById(`step-${step}`);
            
            gsap.to(outEl, { 
                opacity: 0, 
                scale: 0.8, 
                y: -50, 
                duration: 0.5, 
                ease: "power2.in",
                onComplete: () => {
                    outEl.classList.remove('active-card');
                    inEl.classList.add('active-card');
                    gsap.fromTo(inEl, 
                        { opacity: 0, scale: 0.8, y: 50 },
                        { opacity: 1, scale: 1, y: 0, duration: 0.8, ease: "back.out(1.7)" }
                    );
                    // Stagger internal items
                    gsap.from(inEl.querySelectorAll('h1, p, button, .bento-grid, .polaroid-container'), {
                        opacity: 0, y: 20, stagger: 0.1, duration: 0.5, delay: 0.2
                    });
                }
            });

            currentStep = step;
            document.getElementById('progress-bar').style.width = `${(step / 5) * 100}%`;
        }

        // --- POLAROID PARALLAX ---
        const pol = document.getElementById('polaroid');
        document.addEventListener('mousemove', (e) => {
            if(currentStep !== 4) return;
            const x = (window.innerWidth / 2 - e.pageX) / 20;
            const y = (window.innerHeight / 2 - e.pageY) / 20;
            pol.style.transform = `rotateY(${x}deg) rotateX(${y}deg) rotateZ(-3deg)`;
        });

        // --- FINALE ---
        function startFinale() {
            const btn = document.getElementById('celebrate-btn');
            const finaleText = document.getElementById('finale-text');
            
            btn.disabled = true;
            gsap.to(btn, { opacity: 0, scale: 0, duration: 0.5 });
            
            setTimeout(() => {
                finaleText.classList.remove('hidden');
                gsap.from(finaleText, { y: 50, opacity: 0, duration: 1, ease: "power4.out" });
            }, 600);

            // Confetti
            const end = Date.now() + (5 * 1000);
            const colors = ['#ec4899', '#ef4444', '#ffffff', '#ffd700'];

            (function frame() {
                confetti({
                    particleCount: 3,
                    angle: 60,
                    spread: 55,
                    origin: { x: 0, y: 0.8 },
                    colors: colors
                });
                confetti({
                    particleCount: 3,
                    angle: 120,
                    spread: 55,
                    origin: { x: 1, y: 0.8 },
                    colors: colors
                });

                if (Date.now() < end) {
                    requestAnimationFrame(frame);
                }
            }());

            // Emoji Burst
            confetti({
                particleCount: 40,
                spread: 100,
                origin: { y: 0.6 },
                shapes: ['circle'],
                scalar: 2,
                flat: true
            });

            // Heart Flight
            hearts.forEach((h, i) => {
                gsap.to(h.position, {
                    x: (Math.random() - 0.5) * 50,
                    y: 20,
                    z: (Math.random() - 0.5) * 20,
                    duration: 4,
                    delay: i * 0.05,
                    ease: "power2.in"
                });
                gsap.to(h.material, { opacity: 0, duration: 3, delay: 1 });
            });
        }

        window.addEventListener('resize', () => {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        });

        // Start
        initThree();
        animate();
    </script>
